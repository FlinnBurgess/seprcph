199 def setup_file_logger(filename, formatting, log_level):
200  """
201  A helper function for creating a file logger.
202 
203  Args:
204  filename: The file that shall be logged to.
205  formatting: A tuple containing the log format and then the date format.
206  log_level: The level of logging to be reported.
207 
208  Returns:
209  logger: A complete logger object.
210  """
211  logger = logging.getLogger()
212  # If a stream handler has been attached, remove it.
213  if logger.handlers:
214  logger.removeHandler(logger.handlers[0])
215  handler = logging.FileHandler(filename)
216  logger.addHandler(handler)
217  formatter = logging.Formatter(*formatting)
218  handler.setFormatter(formatter)
219  logger.setLevel(log_level)
220  handler.setLevel(log_level)
221  return logger
