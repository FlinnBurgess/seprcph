161 class Container(Element):
162  """
163  A container class which will contain other UI elements
164  """
165  def __init__(self, size, pos, image=None):
166  """
167  Create the container.
168 
169  Args:
170  size: a tuple containing the height and width of the UI element
171  pos: a tuple containing the coordinates of the UI element
172  image: A pygame surface (not required)
173  """
174  if not image:
175  image = pygame.Surface(size, pygame.SRCALPHA, 32)
176  image = image.convert()
177  super(Container, self).__init__(size, pos, image)
178  self.elems = pygame.sprite.LayeredUpdates()
179 
180  def add(self, element):
181  """
182  Add an element to the container
183 
184  Args:
185  element: The element to be added to the container
186  """
187  element.pos = (element.pos[0] + self.pos[0],
188  element.pos[1] + self.pos[1])
189 
190  if not pygame.sprite.collide_rect(self, element) or \
191  element.rect.topleft < self.rect.topleft or \
192  element.rect.bottomright > self.rect.bottomright:
193  raise OutsideContainerError("element %s is outside of container %s",
194  str(element), str(self))
195 
196  element.layer = self.layer + 1
197  self.elems.add(element, layer=element.layer)
198 
199  def remove(self, element):
200  """
201  Remove an element from the container
202 
203  Args:
204  element: The element to be removed from the container
205  """
206  self.elems.remove(element)

208  def click(self, event):
209  """
210  Creates a list of UI elements clicked and executes their callbacks.
211 
212  Args:
213  event: The Event resulting from a mouse click.
214  """
215  for elem in [e for e in self.elems if e.rect.collidepoint(event.pos)]:
216  try:
217  elem.click(event)
218  except AttributeError:
219  pass

221  def resize(self, size):
222  """
223  Called when the main Pygame screen is resized
224 
225  Args:
226  size: The new size of the container
227  """
228  w_ratio = float(size[0]) / float(self.size[0])
229  h_ratio = float(size[1]) / float(self.size[1])
230  for elem in self.elems:
231  elem.resize((int(elem.size[0] * w_ratio), int(elem.size[1] * h_ratio)))
232 
233  self.size = (int(self.size[0] * w_ratio), int(self.size[1] * h_ratio))

272  def add(self, element):
273  """
274  Add an element to the Window
275 
276  Args:
277  element: The element to be added.
278 
279  We are adding all of the sprites into a single sprite group -
280  "flattening out" the sprites. The correct way to do this is
281  to recursively visiti each container, rendering its elements as we go.
282  This is because the Containers work like trees.
283 
284  + Window +
285  | |
286  | |
287  + +
288  Container Container
289  + + +
290  + + +
291  Clickable Label Container
292  """
293  element.layer += self.layer
294  if isinstance(element, Container):
295  for elem in element.elems:
296  elem.layer += self.layer
297  self.elems.add(element.elems, layer=element.layer)
298  self.elems.add(element, layer=element.layer)
